Const createWindow

  • createWindow(url?: string, width?: number, height?: number, opt?: CreateWinOpt): Promise<any>

  • 创建原生窗口

    引入版本

    v0.0.1

    Parameters

    • Optional url: string

      新窗口要加载的url

    • Optional width: number

      新窗口宽度

    • Optional height: number

      新窗口高度

    • Optional opt: CreateWinOpt

      新窗口配置选项

    Returns Promise<any>

    import { native } from '@peiyou/eevee-helper';
// 创建普通窗口
native.createWindow('http://localhost:8080/index.html#/mainPage', 1366, 768, {
     x: 0,
     y: 0,
     router: 'mainPage',
})

// 创建父子窗口
native.createWindow('http://localhost:8080/index.html#/mainPage', 1366, 768, {
     x: 0,
     y: 0,
     router: 'mainPage',
     color: '#000000', // 穿透颜色
     parentId: 1, // 父窗口id
})

Const initWindowSize

  • initWindowSize(width: number, height: number, isFixed?: boolean): Promise<any>

    引入版本

    v0.0.1

    Parameters

    • width: number

      窗口宽度

    • height: number

      窗口高度

    • Optional isFixed: boolean

      是否固定宽高（禁止缩放），默认false

    Returns Promise<any>

native.initWindowSize(1366, 768, true)
